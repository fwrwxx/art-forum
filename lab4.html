Розширені можливості Node.js-додатків: логування, завантаження файлів, моніторинг продуктивності

Дисципліна: WEB-орієнтовані технології. Backend розробки

Студентка: Штифлюк Ольга Володимирівна

🎯 Мета роботи

  • Ознайомитися з розширеними можливостями серверних застосунків на базі Node.js

2. Налаштування Multer для завантаження файлів

// middleware/uploadMiddleware.js
const multer = require('multer');

const storage = multer.diskStorage({
    destination: (req, file, cb) => {
        let subDir = file.mimetype.startsWith('image/') ? 'images' : 'documents';
        cb(null, path.join('uploads', subDir));
    },
    filename: (req, file, cb) => {
        const uniqueSuffix = Date.now() + '-' + Math.round(Math.random() * 1E9);
        cb(null, `${uniqueSuffix}-${file.originalname}`);
    }
});

const upload = multer({
    storage,
    limits: { fileSize: 5 * 1024 * 1024 },
    fileFilter: (req, file, cb) => {
        const allowedTypes = ['image/jpeg', 'image/png', 'application/pdf'];
        if (allowedTypes.includes(file.mimetype)) cb(null, true);
        else cb(new Error('File type not allowed'), false);
    }
});

3. Вимірювання продуктивності (Performance Middleware)

// middleware/performanceMiddleware.js
const performanceMiddleware = (req, res, next) => {
    const start = Date.now();
    
    res.on('finish', () => {
        const duration = Date.now() - start;
        logger.info(`Performance: ${req.method} ${req.url} - ${duration}ms`);
        res.setHeader('X-Response-Time', `${duration}ms`);
    });
    
    next();
};

const memoryMonitorMiddleware = (req, res, next) => {
    const memoryUsage = process.memoryUsage();
    const heapUsedMB = memoryUsage.heapUsed / 1024 / 1024;
    
    if (heapUsedMB > 500) {
        console.warn(`High memory usage: ${heapUsedMB.toFixed(2)} MB`);
    }
    next();
};

4. Моніторинг стану сервера

// routes/statusRoutes.js
router.get('/', (req, res) => {
    const memoryUsage = process.memoryUsage();
    const cpuUsage = process.cpuUsage();
    const uptime = process.uptime();
    
    res.json({
        uptime: { seconds: uptime, formatted: formatUptime(uptime) },
        memory: {
            heapUsed: `${(memoryUsage.heapUsed / 1024 / 1024).toFixed(2)} MB`,
            heapTotal: `${(memoryUsage.heapTotal / 1024 / 1024).toFixed(2)} MB`,
            rss: `${(memoryUsage.rss / 1024 / 1024).toFixed(2)} MB`
        },
        cpu: {
            user: `${(cpuUsage.user / 1000).toFixed(2)} ms`,
            system: `${(cpuUsage.system / 1000).toFixed(2)} ms`
        }
    });
});

⚙️ Управління застосунком через PM2

# Встановлення PM2 глобально
npm install -g pm2

# Запуск застосунку
pm2 start app.js --name lab4-app

# Запуск з конфігураційним файлом
pm2 start ecosystem.config.js

# Список процесів
pm2 list

# Моніторинг в реальному часі
pm2 monit

# Перегляд логів
pm2 logs lab4-app
pm2 logs lab4-app --lines 100

# Рестарт та зупинка
pm2 restart lab4-app
pm2 stop lab4-app
pm2 delete lab4-app

# Збереження конфігурації для автозапуску
pm2 save
pm2 startup
                    
// ecosystem.config.js
module.exports = {
    apps: [{
        name: 'lab4-app',
        script: 'app.js',
        instances: 1,

        env: { NODE_ENV: 'production', PORT: 3000 },
        error_file: './logs/pm2-error.log',

    }]
};
